33	/**
34	 * <https://vulkan-tutorial.com/Drawing_a_triangle/Setup/Physical_devices_and_queue_families>
35	 * this will get the first graphics card avaliable
36	 */
37	fn get_device(instance: &ash::Instance) -> Result<ash::Device, Box<dyn Error>>
38	{
39		/**
40		 * C++
41		 * uint32_t deviceCount = 0;
42		 * vkEnumeratePhysicalDevices(instance, &deviceCount, nullptr);
43		 * same idea from ours, but ash offers less control, no need to implement verifications of support
44		 */
45		let physical_devices = unsafe { instance.enumerate_physical_devices()? };
46		let physical_device = physical_devices[0];
47
48		/**
49		 * this can be null if the function enumerate_physical_devices() didn't find any compatible gpu
50		 */
51		if physical_device.is_null() == true || !Self::is_device_suitable(physical_device)
52			{ panic!("failed to find a suitable GPU!"); }
53
54		let queue_families =
55			unsafe { instance.get_physical_device_queue_family_properties(physical_device) };
56
57		let graphics_queue_index = queue_families
58			.iter()
59			.enumerate()
60			.find(|(_, q)| q.queue_flags.contains(vk::QueueFlags::GRAPHICS))
61			.map(|(i, _)| i)
62			.expect("no graphics queue found") as u32;
63
64		let queue_priority = 1.0;
65		let binding = [queue_priority];
66
67		let queue_info = vk::DeviceQueueCreateInfo::default()
68			.queue_family_index(graphics_queue_index)
69			.queue_priorities(&binding);
70
71		let device_features = vk::PhysicalDeviceFeatures::default();
72
73		let device_create_info = vk::DeviceCreateInfo::default()
74			.queue_create_infos(std::slice::from_ref(&queue_info))
75			.enabled_features(&device_features);
76
77		let device_extensions = [
78			vk::KHR_SWAPCHAIN_NAME.as_ptr(),
79		];
80
81		let device_create_info = device_create_info
82			.enabled_extension_names(&device_extensions);
83
84		Ok(
85			unsafe { instance.create_device(physical_device, &device_create_info, None)? }
86		)
87	}

90	fn new_surface(instance: &Instance, entry: &ash::Entry, window: NonNull<c_void>) -> SurfaceKHR
91	{
92		use objc2::{rc::Retained, msg_send, ClassType};
93		use ash::ext::metal_surface;
94		use objc2_quartz_core::CALayer;
95		use objc2_foundation::NSObject;
96		debug!("creating metal surface");
97
98		let ns_view: &NSObject = unsafe { window.cast().as_ref() };
99		let _: () = unsafe { msg_send![ns_view, setWantsLayer: true] };
100
101		let layer: Option<Retained<CALayer>> = unsafe { msg_send![ns_view, layer] };
102		let layer = to_handle(
103			&mut layer.expect("failed making the view layer-backed").as_super()
104		);
105
106		let surface_desc = vk::MetalSurfaceCreateInfoEXT::default().layer(layer);
107		let surface = metal_surface::Instance::new(entry, instance);
108		unsafe {
109			surface.create_metal_surface(&surface_desc, None)
110				.expect("couldn't create metal surface")
111		}
112	}

141	/// Gets what it's needed to the renderer work
142	/// For example, MacOS with `EXT_METAL_SURFACE_NAME`, because it doesn't have a native vulkan renderer
143	fn detect_needed_extensions() -> Vec<*const i8>
144	{Vec::from([
145		vk::KHR_PORTABILITY_ENUMERATION_NAME.as_ptr(),

150	/// Creates a new Vulkan render
151	/// this will be our initVulkan() from the tutorial
152	/// <https://vulkan-tutorial.com/Drawing_a_triangle/Setup/Base_code#:~:text=initVulkan()>
153	pub fn new(surface_backend: *mut c_void) -> Result<Renderer, Box<dyn Error>>
154	{
155		//let surface_backed = surface_backend as *mut SurfaceBackend<Wrapper>;
156		debug!("Creating new vulkan render on backend:\n{:#?}", surface_backend);
157
158		/**
159		 * load vulkan in execution, otherwise one might have a problem compiling it for macos (apple beeing apple)
160		 * another problem is the inexistence of a vulkan dylib natively on mac
161		 * the solution for that problem is packaging the necessary files (dylib) inside the .app
162		 * <https://stackoverflow.com/questions/39204908/how-to-check-release-debug-builds-using-cfg-in-rust>
163		 */
164		let entry = unsafe { ash::Entry::load()? };
165
166		/**
167		 * Create Instance
168		 * <https://vulkan-tutorial.com/en/Drawing_a_triangle/Setup/Instance>
169		 * VkApplicationInfo appInfo{};
170		 */
171		let app_info =
172			vk::ApplicationInfo::default().api_version(vk::make_api_version(0, 1, 0, 0));
173
174		let extensions = Self::detect_needed_extensions();
175		let instance_desc = vk::InstanceCreateInfo::default()
176			.application_info(&app_info)
177			.enabled_extension_names(&extensions);
178
179		let instance: Instance = unsafe {
180			/**
181			 * vkCreateInstance(&createInfo, nullptr, &instance) -> function from C++
182			 * in this case, createInfo is instance_desc
183			 * the Pointer to the variable that stores the handle to the new object (&instance) value isn't necessary
184			 *
185			 * VK_ERROR_INCOMPATIBLE_DRIVER:
186			 * <https://vulkan-tutorial.com/en/Drawing_a_triangle/Setup/Instance#:~:text=Encountered%20VK%5FERROR%5FINCOMPATIBLE%5FDRIVER>
187			 */
188			entry.create_instance(&instance_desc, None)?
189		};
190
191		/**
192		 * Handlers
193		 */
194		let device = Self::get_device(&instance)?;
195		let renderpass = Self::render_pass(&device)?;
196
197		/** <https://github.com/ash-rs/ash/blob/master/ash-examples/src/lib.rs>
198		 * The Headless backend will be used to implement tests
199		 * the repo uses `let surface_loader = surface::Instance::load(&entry, &instance);`
200		 * but the way it is created is different, using `SurfaceFactory`, for that I would need winit
201		 */
202
203		let view = Wrapper::get_surface(surface_backend);
204		let nn_view = NonNull::new(view)
205			.expect("NSView shouldn't be null")
206			.cast();
207
208		let surface = Self::new_surface(&instance, &entry, nn_view);
209
210		Ok(Renderer {
211			instance,
212			surface,
213			device,
214			renderpass,
215		})
216	}

218	/// Creates a new vulkan renderpass
219	/// here's an oficial example: <https://github.com/ash-rs/ash/blob/master/ash-examples/src/bin/texture.rs>
220	pub fn render_pass(device: &ash::Device) -> Result<RenderPass, Box<dyn Error>>
221	{
222		// tbh, I have no idea what does this do
223		let renderpass_attachments = [
224			vk::AttachmentDescription {
225				samples: vk::SampleCountFlags::TYPE_1,
226				load_op: vk::AttachmentLoadOp::CLEAR,
227				store_op: vk::AttachmentStoreOp::STORE,
228				final_layout: vk::ImageLayout::PRESENT_SRC_KHR,
229				..Default::default()
230			},
231			vk::AttachmentDescription {
232				format: vk::Format::D16_UNORM,
233				samples: vk::SampleCountFlags::TYPE_1,
234				load_op: vk::AttachmentLoadOp::CLEAR,
235				initial_layout: vk::ImageLayout::DEPTH_STENCIL_ATTACHMENT_OPTIMAL,
236				final_layout: vk::ImageLayout::DEPTH_STENCIL_ATTACHMENT_OPTIMAL,
237				..Default::default()
238			},
239		];
240
241		let color_attachment_refs = [vk::AttachmentReference {
242			attachment: 0,
243			layout: vk::ImageLayout::COLOR_ATTACHMENT_OPTIMAL,
244		}];
245		let depth_attachment_ref = vk::AttachmentReference {
246			attachment: 1,
247			layout: vk::ImageLayout::DEPTH_STENCIL_ATTACHMENT_OPTIMAL,
248		};
249		let dependencies = [vk::SubpassDependency {
250			src_subpass: vk::SUBPASS_EXTERNAL,
251			src_stage_mask: vk::PipelineStageFlags::COLOR_ATTACHMENT_OUTPUT,
252			dst_access_mask: vk::AccessFlags::COLOR_ATTACHMENT_READ
253				| vk::AccessFlags::COLOR_ATTACHMENT_WRITE,
254			dst_stage_mask: vk::PipelineStageFlags::COLOR_ATTACHMENT_OUTPUT,
255			..Default::default()
256		}];
257
258		let subpass = vk::SubpassDescription::default()
259			.color_attachments(&color_attachment_refs)
260			.depth_stencil_attachment(&depth_attachment_ref)
261			.pipeline_bind_point(vk::PipelineBindPoint::GRAPHICS);
262
263		let renderpass_create_info = vk::RenderPassCreateInfo::default()
264			.attachments(&renderpass_attachments)
265			.subpasses(std::slice::from_ref(&subpass))
266			.dependencies(&dependencies);
267
268		let renderpass = unsafe {
269			device
270				.create_render_pass(&renderpass_create_info, None)
271				.unwrap()
272		};
273
274		return Ok(renderpass);
275	}
