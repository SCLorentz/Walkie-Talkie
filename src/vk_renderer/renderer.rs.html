35	/**
36	 * <https://vulkan-tutorial.com/Drawing_a_triangle/Setup/Physical_devices_and_queue_families>
37	 * this will get the first graphics card avaliable
38	 */
39	fn get_device(instance: &ash::Instance) -> Result<ash::Device, Box<dyn Error>>
40	{
41		/**
42		 * C++
43		 * uint32_t deviceCount = 0;
44		 * vkEnumeratePhysicalDevices(instance, &deviceCount, nullptr);
45		 * same idea from ours, but ash offers less control, no need to implement verifications of support
46		 */
47		let physical_devices = unsafe { instance.enumerate_physical_devices()? };
48		let physical_device = physical_devices[0];
49
50		/**
51		 * this can be null if the function enumerate_physical_devices() didn't find any compatible gpu
52		 */
53		if physical_device.is_null() == true || !Self::is_device_suitable(physical_device)
54		{
55			panic!("failed to find a suitable GPU!");
56		}
57
58		let queue_families =
59			unsafe { instance.get_physical_device_queue_family_properties(physical_device) };
60
61		let graphics_queue_index = queue_families
62			.iter()
63			.enumerate()
64			.find(|(_, q)| q.queue_flags.contains(vk::QueueFlags::GRAPHICS))
65			.map(|(i, _)| i)
66			.expect("no graphics queue found") as u32;
67
68		let queue_priority = 1.0;
69		let binding = [queue_priority];
70
71		let queue_info = vk::DeviceQueueCreateInfo::default()
72			.queue_family_index(graphics_queue_index)
73			.queue_priorities(&binding);
74
75		let device_features = vk::PhysicalDeviceFeatures::default();
76
77		let device_create_info = vk::DeviceCreateInfo::default()
78			.queue_create_infos(std::slice::from_ref(&queue_info))
79			.enabled_features(&device_features);
80
81		let device_extensions = [
82			vk::KHR_SWAPCHAIN_NAME.as_ptr(),
83		];
84
85		let device_create_info = device_create_info
86			.enabled_extension_names(&device_extensions);
87
88		Ok(
89			unsafe { instance.create_device(physical_device, &device_create_info, None)? }
90		)
91	}

143	#[cfg(target_os = "macos")]
144	fn new_surface(instance: &Instance, entry: &ash::Entry, window: NonNull<c_void>) -> SurfaceKHR
145	{
146		use objc2::{rc::Retained, msg_send, ClassType};
147		use ash::ext::metal_surface;
148		use objc2_quartz_core::CALayer;
149		use objc2_foundation::NSObject;
150		debug!("creating metal surface");
151
152		let ns_view: &NSObject = unsafe { window.cast().as_ref() };
153		let _: () = unsafe { msg_send![ns_view, setWantsLayer: true] };
154
155		let layer: Option<Retained<CALayer>> = unsafe { msg_send![ns_view, layer] };
156		let layer = to_handle(
157			&mut layer.expect("failed making the view layer-backed").as_super()
158		);
159
160		let surface_desc = vk::MetalSurfaceCreateInfoEXT::default().layer(layer);
161		let surface = metal_surface::Instance::new(entry, instance);
162		unsafe {
163			surface.create_metal_surface(&surface_desc, None)
164				.expect("couldn't create metal surface")
165		}
166	}

180		let window: &WaylandWindowHandle = unsafe { window.cast().as_ref() };
181		/**
182		 * https://docs.rs/ash-window/0.13.0/src/ash_window/lib.rs.html#36-126

184		let display = std::ptr::null_mut() as *const c_void as *mut wl_display;
185
186		let surface_desc = vk::WaylandSurfaceCreateInfoKHR::default()
187			.display(display)
188			.surface((*window).surface.as_ptr());
189
190		let surface = wayland_surface::Instance::new(entry, instance);
191		unsafe { surface.create_wayland_surface(&surface_desc, None)
192			.expect("couldn't create wayland surface") }
193	}

195	/// Gets what it's needed to the renderer work
196	/// For example, MacOS with `EXT_METAL_SURFACE_NAME`, because it doesn't have a native vulkan renderer
197	fn detect_needed_extensions() -> Vec<*const i8>
198	{Vec::from([
199		vk::KHR_PORTABILITY_ENUMERATION_NAME.as_ptr(),
200		vk::KHR_SURFACE_NAME.as_ptr(),
201		vk::EXT_METAL_SURFACE_NAME.as_ptr(),

204	/// Creates a new Vulkan render
205	/// this will be our initVulkan() from the tutorial
206	/// <https://vulkan-tutorial.com/Drawing_a_triangle/Setup/Base_code#:~:text=initVulkan()>
207	pub fn new(surface_backend: &mut SurfaceBackend) -> Result<Renderer, Box<dyn Error>>
208	{
209		// for some reason idk how to take this off without invalid memory address
210		let ptr = surface_backend as *mut SurfaceBackend;
211		let surface_backend = unsafe { (*ptr).clone() };
212		debug!("Creating new vulkan render on backend:\n{:#?}", surface_backend);
213
214		/**
215		 * load vulkan in execution, otherwise one might have a problem compiling it for macos (apple beeing apple)
216		 * another problem is the inexistence of a vulkan dylib natively on mac
217		 * the solution for that problem is packaging the necessary files (dylib) inside the .app
218		 * <https://stackoverflow.com/questions/39204908/how-to-check-release-debug-builds-using-cfg-in-rust>
219		 */
220		let entry = unsafe { ash::Entry::load()? };
221
222		/**
223		 * Create Instance
224		 * <https://vulkan-tutorial.com/en/Drawing_a_triangle/Setup/Instance>
225		 * VkApplicationInfo appInfo{};
226		 */
227		let app_info =
228			vk::ApplicationInfo::default().api_version(vk::make_api_version(0, 1, 0, 0));
229
230		let extensions = Self::detect_needed_extensions();
231		let instance_desc = vk::InstanceCreateInfo::default()
232			.application_info(&app_info)
233			.enabled_extension_names(&extensions);
234
235		let instance: Instance = unsafe {
236			/**
237			 * vkCreateInstance(&createInfo, nullptr, &instance) -> function from C++
238			 * in this case, createInfo is instance_desc
239			 * the Pointer to the variable that stores the handle to the new object (&instance) value isn't necessary
240			 *
241			 * VK_ERROR_INCOMPATIBLE_DRIVER:
242			 * <https://vulkan-tutorial.com/en/Drawing_a_triangle/Setup/Instance#:~:text=Encountered%20VK%5FERROR%5FINCOMPATIBLE%5FDRIVER>
243			 */
244			entry.create_instance(&instance_desc, None)?
245		};
246
247		/**
248		 * Handlers
249		 */
250		let device = Self::get_device(&instance)?;
251		let renderpass = Self::render_pass(&device)?;
252
253		/** <https://github.com/ash-rs/ash/blob/master/ash-examples/src/lib.rs>
254		 * The Headless backend will be used to implement tests
255		 * the repo uses `let surface_loader = surface::Instance::load(&entry, &instance);`
256		 * but the way it is created is different, using `SurfaceFactory`, for that I would need winit
257		 */
258
259		let view = match surface_backend {
260			#[cfg(debug_assertions)]
261			SurfaceBackend::Headless => todo!(),
262			SurfaceBackend::MacOS(surface) => surface.ns_view,
263			SurfaceBackend::Linux { .. } => todo!(),
264			_ => todo!()
265		};
266
267		let nn_view = NonNull::new(view)
268			.expect("NSView shouldn't be null")
269			.cast();
270
271		let surface = Self::new_surface(&instance, &entry, nn_view);
272
273		Ok(Renderer {
274			instance,
275			surface,
276			device,
277			renderpass,
278		})
279	}

281	/// Creates a new vulkan renderpass
282	/// here's an oficial example: <https://github.com/ash-rs/ash/blob/master/ash-examples/src/bin/texture.rs>
283	pub fn render_pass(device: &ash::Device) -> Result<RenderPass, Box<dyn Error>>
284	{
285		// tbh, I have no idea what does this do
286		let renderpass_attachments = [
287			vk::AttachmentDescription {
288				samples: vk::SampleCountFlags::TYPE_1,
289				load_op: vk::AttachmentLoadOp::CLEAR,
290				store_op: vk::AttachmentStoreOp::STORE,
291				final_layout: vk::ImageLayout::PRESENT_SRC_KHR,
292				..Default::default()
293			},
294			vk::AttachmentDescription {
295				format: vk::Format::D16_UNORM,
296				samples: vk::SampleCountFlags::TYPE_1,
297				load_op: vk::AttachmentLoadOp::CLEAR,
298				initial_layout: vk::ImageLayout::DEPTH_STENCIL_ATTACHMENT_OPTIMAL,
299				final_layout: vk::ImageLayout::DEPTH_STENCIL_ATTACHMENT_OPTIMAL,
300				..Default::default()
301			},
302		];
303
304		let color_attachment_refs = [vk::AttachmentReference {
305			attachment: 0,
306			layout: vk::ImageLayout::COLOR_ATTACHMENT_OPTIMAL,
307		}];
308		let depth_attachment_ref = vk::AttachmentReference {
309			attachment: 1,
310			layout: vk::ImageLayout::DEPTH_STENCIL_ATTACHMENT_OPTIMAL,
311		};
312		let dependencies = [vk::SubpassDependency {
313			src_subpass: vk::SUBPASS_EXTERNAL,
314			src_stage_mask: vk::PipelineStageFlags::COLOR_ATTACHMENT_OUTPUT,
315			dst_access_mask: vk::AccessFlags::COLOR_ATTACHMENT_READ
316				| vk::AccessFlags::COLOR_ATTACHMENT_WRITE,
317			dst_stage_mask: vk::PipelineStageFlags::COLOR_ATTACHMENT_OUTPUT,
318			..Default::default()
319		}];
320
321		let subpass = vk::SubpassDescription::default()
322			.color_attachments(&color_attachment_refs)
323			.depth_stencil_attachment(&depth_attachment_ref)
324			.pipeline_bind_point(vk::PipelineBindPoint::GRAPHICS);
325
326		let renderpass_create_info = vk::RenderPassCreateInfo::default()
327			.attachments(&renderpass_attachments)
328			.subpasses(std::slice::from_ref(&subpass))
329			.dependencies(&dependencies);
330
331		let renderpass = unsafe {
332			device
333				.create_render_pass(&renderpass_create_info, None)
334				.unwrap()
335		};
336
337		return Ok(renderpass);
338	}
