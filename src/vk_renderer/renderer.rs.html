54	/// Creates a new Vulkan render
55	/// this will be our `initVulkan()` from the tutorial
56	/// <https://vulkan-tutorial.com/Drawing_a_triangle/Setup/Base_code#:~:text=initVulkan()>
57	pub fn new(surface_backend: *mut void) -> Result<Renderer, Box<dyn Error>>
58	{
59		let backend: Wrapper = void::from_handle(surface_backend);
60		debug!("Creating new vulkan render");
61
62		/**
63		 * load vulkan in execution, otherwise one might have a problem compiling it for macos (apple beeing apple)
64		 * another problem is the inexistence of a vulkan dylib natively on mac
65		 * the solution for that problem is packaging the necessary files (dylib) inside the .app
66		 * <https://stackoverflow.com/questions/39204908/how-to-check-release-debug-builds-using-cfg-in-rust>
67		 */
68		let entry = unsafe { ash::Entry::load()? };
69
70		/**
71		 * Create Instance
72		 * <https://vulkan-tutorial.com/en/Drawing_a_triangle/Setup/Instance>
73		 * `VkApplicationInfo appInfo{};`
74		 * for some reason on target linux-a64 it expects "u8" and not "i8" idk y
75		 */
76		let app_info = vk::ApplicationInfo::default()
77			.api_version(vk::make_api_version(0, 1, 0, 0));
78
79		let extensions: &[*const f8] = &[
80			vk::KHR_PORTABILITY_ENUMERATION_NAME.as_ptr(),
81			vk::KHR_SURFACE_NAME.as_ptr(),
82			#[cfg(target_os = "macos")]
83			vk::EXT_METAL_SURFACE_NAME.as_ptr()
84		];
85
86		/* */
87
88		let instance_desc = vk::InstanceCreateInfo::default()
89			.application_info(&app_info)
90			.enabled_extension_names(extensions);
91
92		let instance: Instance = unsafe {
93			/**
94			 * `vkCreateInstance(&createInfo, nullptr, &instance)` -> function from C++
95			 * in this case, createInfo is `instance_desc`
96			 * the Pointer to the variable that stores the handle to the new object (&instance) value isn't necessary
97			 *
98			 * `VK_ERROR_INCOMPATIBLE_DRIVER`:
99			 * <https://vulkan-tutorial.com/en/Drawing_a_triangle/Setup/Instance#:~:text=Encountered%20VK%5FERROR%5FINCOMPATIBLE%5FDRIVER>
100			 */
101			entry.create_instance(&instance_desc, None)?
102		};
103
104		/**
105		 * Handlers
106		 */
107		let device = Self::get_device(&instance)?;
108		//let _logic = Self::create_logic_device();
109		let renderpass = Self::render_pass(&device)?;
110
111		/** <https://github.com/ash-rs/ash/blob/master/ash-examples/src/lib.rs>
112		 * The Headless backend will be used to implement tests
113		 * the repo uses `let surface_loader = surface::Instance::load(&entry, &instance);`
114		 * but the way it is created is different, using `SurfaceFactory`, for that I would need winit
115		 */
116
117		#[cfg(target_os = "macos")]
118		let view = backend.ns_view;
119
120		#[cfg(target_os = "linux")]
121		let view: *mut void = backend.surface;
122
123		#[cfg(target_os = "windows")]
124		let view: *mut void = todo!();
125
126		let Some(nn_view) = NonNull::new(view) else {
127			return Err(Box::from("view shouldn't be null"))
128		};
129
130		let surface = Self::new_surface(&instance, &entry, nn_view.cast())?;
131
132		Ok(Renderer {
133			surface,
134			renderpass,
135			device,
136			instance,
137		})
138	}

144	/**
145	 * in the future, implement a rate to pick the best device avaliable
146	 * get the device properties and check it can be used
147	 * <https://docs.vulkan.org/tutorial/latest/03_Drawing_a_triangle/00_Setup/03_Physical_devices_and_queue_families.html>
148	 * <https://vulkan-tutorial.com/Drawing_a_triangle/Setup/Physical_devices_and_queue_families#:~:text=isDeviceSuitable>
149	 */
150	#[inline]
151	fn is_device_suitable(instance: &ash::Instance, device: PhysicalDevice) -> bool
152	{
153		let prop = unsafe { instance.get_physical_device_properties(device) };
154		let feat = unsafe { instance.get_physical_device_features(device) };
155
156		/**
157		 * `INTEGRATED_GPU` will be the case on `MacOS`
158		 * for some reason the constant value for that is not working, so im using `.as_raw() == 1`
159		 * "associated item not found in `PhysicalDevice`"
160		 */
161		if feat.geometry_shader == dirty::TRUE
162			&& prop.device_type == vk::PhysicalDeviceType::DISCRETE_GPU
163			|| prop.device_type.as_raw() == 1 // this represents Self(1) or INTEGRATED_GPU
164		{ return true; }
165
166		false
167	}

169	/**
170	 * <https://vulkan-tutorial.com/Drawing_a_triangle/Setup/Physical_devices_and_queue_families>
171	 * this will get the first graphics card avaliable
172	 */
173	fn get_device(instance: &ash::Instance) -> Result<ash::Device, Box<dyn Error>>
174	{
175		/**
176		 * C++
177		 * `uint32_t deviceCount = 0;`
178		 * `vkEnumeratePhysicalDevices(instance, &deviceCount, nullptr);`
179		 * same idea from ours, but ash offers less control, no need to implement verifications of support
180		 */
181		let physical_devices = unsafe { instance.enumerate_physical_devices()? };
182
183		let mut maybe_selected_device: Option<PhysicalDevice> = None;
184		for device in physical_devices {
185			if !Self::is_device_suitable(instance, device) { continue }
186			debug!("found suitable device `{device:?}`!");
187			maybe_selected_device = Some(device);
188			break
189		}
190
191		let Some(selected_device) = maybe_selected_device else {
192			return Err(Box::from("failed to find a suitable GPU!"))
193		};
194
195		let queue_families = unsafe {
196			instance.get_physical_device_queue_family_properties(selected_device)
197		};
198
199		let graphics_queue_index: u32 = u32::try_from(
200			queue_families
201				.iter()
202				.enumerate()
203				.find(|(_, q)| q.queue_flags.contains(vk::QueueFlags::GRAPHICS))
204				.ok_or("no graphics queue found")?
205				.0
206		)?;
207
208		let queue_priority = 1.0;
209		let binding = [queue_priority];
210
211		let queue_info = vk::DeviceQueueCreateInfo::default()
212			.queue_family_index(graphics_queue_index)
213			.queue_priorities(&binding);
214
215		let device_features = vk::PhysicalDeviceFeatures::default();
216
217		let device_extensions = [
218			vk::KHR_SWAPCHAIN_NAME.as_ptr(),
219		];
220
221		let device_create_info = vk::DeviceCreateInfo::default()
222			.enabled_extension_names(&device_extensions)
223			.queue_create_infos(slice::from_ref(&queue_info))
224			.enabled_features(&device_features);
225
226		Ok(unsafe {
227			instance.create_device(selected_device, &device_create_info, None)?
228		})
229	}

231	/// Creates a new surface
232	#[cfg(target_os = "macos")]
233	fn new_surface(
234		instance: &Instance,
235		entry: &ash::Entry,
236		window: NonNull<void>
237	) -> Result<SurfaceKHR, Box<dyn Error>>
238	{
239		use objc2::{rc::Retained, msg_send, ClassType};
240		use objc2_quartz_core::CALayer;
241		use objc2_foundation::NSObject;
242		use ash::ext::metal_surface;
243		debug!("creating metal surface");
244
245		let ns_view: &NSObject = void::from_handle(window.as_ptr());
246		let _: () = unsafe { msg_send![ns_view, setWantsLayer: true] };
247
248		let Some(layer_some): Option<Retained<CALayer>> = (
249			unsafe { msg_send![ns_view, layer] }
250		) else {
251			return Err(Box::from("failed making the view layer-backed"))
252		};
253		let layer = void::to_handle(&mut layer_some.as_super());
254
255		let surface_desc = vk::MetalSurfaceCreateInfoEXT::default()
256			.layer(layer.cast::<core::ffi::c_void>());
257		let surface = metal_surface::Instance::new(entry, instance);
258
259		let Ok(metal_surface) = (
260			unsafe { surface.create_metal_surface(&surface_desc, None) }
261		) else {
262			return Err(Box::from("couldn't create metal surface"))
263		};
264
265		Ok(metal_surface)
266	}

269	/// Creates a new surface
270	#[cfg(target_os = "linux")]
271	fn new_surface(
272		instance: &Instance,
273		entry: &ash::Entry,
274		window: NonNull<void>
275	) -> Result<SurfaceKHR, Box<dyn Error>>
276	{
277		debug!("creating linux wayland surface");
278		use ash::{khr::wayland_surface, vk::wl_display};
279
280		let window_ref: &Wrapper = unsafe { window.cast().as_ref() };
281		/**
282		 * https://docs.rs/ash-window/0.13.0/src/ash_window/lib.rs.html#36-126
283		 */
284		let display = core::ptr::null_mut() as *const void as *mut wl_display;
285
286		let surface_desc = vk::WaylandSurfaceCreateInfoKHR::default()
287			.display(display)
288			.surface(window_ref.surface as *mut core::ffi::c_void);
289
290		let surface = wayland_surface::Instance::new(entry, instance);
291		let result = unsafe { surface.create_wayland_surface(&surface_desc, None)? };
292
293		Ok(result)
294	}

307	/// Creates a new vulkan renderpass
308	/// here's an oficial example: <https://github.com/ash-rs/ash/blob/master/ash-examples/src/bin/texture.rs>
309	pub fn render_pass(device: &ash::Device) -> Result<RenderPass, Box<dyn Error>>
310	{
311		// tbh, I have no idea what does this do
312		let renderpass_attachments = [
313			vk::AttachmentDescription {
314				samples: vk::SampleCountFlags::TYPE_1,
315				load_op: vk::AttachmentLoadOp::CLEAR,
316				store_op: vk::AttachmentStoreOp::STORE,
317				final_layout: vk::ImageLayout::PRESENT_SRC_KHR,
318				..Default::default()
319			},
320			vk::AttachmentDescription {
321				format: vk::Format::D16_UNORM,
322				samples: vk::SampleCountFlags::TYPE_1,
323				load_op: vk::AttachmentLoadOp::CLEAR,
324				initial_layout: vk::ImageLayout::DEPTH_STENCIL_ATTACHMENT_OPTIMAL,
325				final_layout: vk::ImageLayout::DEPTH_STENCIL_ATTACHMENT_OPTIMAL,
326				..Default::default()
327			},
328		];
329
330		let color_attachment_refs = [vk::AttachmentReference {
331			attachment: 0,
332			layout: vk::ImageLayout::COLOR_ATTACHMENT_OPTIMAL,
333		}];
334		let depth_attachment_ref = vk::AttachmentReference {
335			attachment: 1,
336			layout: vk::ImageLayout::DEPTH_STENCIL_ATTACHMENT_OPTIMAL,
337		};
338		let dependencies = [vk::SubpassDependency {
339			src_subpass: vk::SUBPASS_EXTERNAL,
340			src_stage_mask: vk::PipelineStageFlags::COLOR_ATTACHMENT_OUTPUT,
341			dst_access_mask: vk::AccessFlags::COLOR_ATTACHMENT_READ
342				| vk::AccessFlags::COLOR_ATTACHMENT_WRITE,
343			dst_stage_mask: vk::PipelineStageFlags::COLOR_ATTACHMENT_OUTPUT,
344			..Default::default()
345		}];
346
347		let subpass = vk::SubpassDescription::default()
348			.color_attachments(&color_attachment_refs)
349			.depth_stencil_attachment(&depth_attachment_ref)
350			.pipeline_bind_point(vk::PipelineBindPoint::GRAPHICS);
351
352		let renderpass_create_info = vk::RenderPassCreateInfo::default()
353			.attachments(&renderpass_attachments)
354			.subpasses(slice::from_ref(&subpass))
355			.dependencies(&dependencies);
356
357		let renderpass = unsafe {
358			match device
359				.create_render_pass(&renderpass_create_info, None) {
360					Ok(d) => d,
361					Err(e) => return Err(Box::new(e)),
362				}
363		};
364
365		Ok(renderpass)
366	}
