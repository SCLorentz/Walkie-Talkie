21fn app_loop(event: Option<Event>)
22{
23	match event {
24		Some(Event::CloseRequest) => info!("closing now"),
25		Some(Event::WindowResized { window: w, .. }) => info!("Resizing window: {:?}", w.title),
26		Some(Event::ThemeChange { new_theme: theme }) => info!("changed: {:?}", theme),
27		Some(other) => info!("event: {:?}", other),
28		None => {}
29	}
