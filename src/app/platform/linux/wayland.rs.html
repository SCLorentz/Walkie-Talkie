26struct State {
27	running: bool,
28	base_surface: Option<*mut c_void>,	// wl_surface::WlSurface
29	buffer:	Option<*mut c_void>,		// wl_buffer::WlBuffer
30	wm_base: Option<*mut c_void>,		// xdg_wm_base::XdgWmBase
31	xdg_surface: Option<*const c_void>,
32	configured: bool,
33	title: String,
34}

36impl State {
37	fn init_xdg_surface(&mut self, qh: &QueueHandle<State>)
38	{
39		let wm_base = self.wm_base.unwrap() as *mut xdg_wm_base::XdgWmBase;
40		let base_surface = self.base_surface.unwrap() as *mut wl_surface::WlSurface;
41
42		let xdg_surface = unsafe { (*wm_base).get_xdg_surface(&*base_surface, qh, ()) };
43		let toplevel = xdg_surface.get_toplevel(qh, ());
44		toplevel.set_title(<String as Clone>::clone(&self.title).into());
45
46		unsafe { (*base_surface).commit() };
47
48		self.xdg_surface =
49			Some(to_handle(WaylandFrame { xdg_surface, toplevel }));
50	}
51}

72// wayland_protocols (which include wayland_client) failed to build documentation on version 0.31.12 thks!!
73impl WaylandDecoration for Decoration
74{
75	fn new(title: String, _width: f64, _height: f64) -> Decoration
76	{
77		let conn = Connection::connect_to_env().unwrap();
78
79		let mut event_queue = conn.new_event_queue();
80		let qhandle = event_queue.handle();
81
82		let display = conn.display();
83		display.get_registry(&qhandle, ());
84
85		let mut state = State {
86			running: true,
87			base_surface: None,
88			buffer: None,
89			wm_base: None,
90			xdg_surface: None,
91			configured: false,
92			title,
93		};
94
95		while state.running {
96			event_queue.blocking_dispatch(&mut state).unwrap();
97		}
98
99		/**
100		 * This version will include SSDs and DBusMenu
101		 * <https://docs.rs/dbusmenu-glib/latest/dbusmenu_glib/>
102		 * On KDE, implement:
103		 * - <https://wayland.app/protocols/kde-blur>
104		 * - <https://wayland.app/protocols/kde-appmenu>
105		 * On Hyprland, implement:
106		 * - <https://wayland.app/protocols/hyprland-surface-v1>
107		 * Other future (optional) implementations may include:
108		 * - popups, notifications, tablet, ext_background_effect_manager_v1
109		 */
110		let backend = LinuxWrapper {
111			state: to_handle(state)
112		};
113
114		return Decoration {
115			mode: DecorationMode::ServerSide,
116			frame: std::ptr::null_mut() as *const c_void, // TODO
117			backend,
118		};
119	}
120
121	fn init_event_state() -> EventQueue<State>
122	{
123		let connection = Connection::connect_to_env().unwrap();
124		connection.new_event_queue()
125	}
126
127	fn make_view() {}
128
129	fn apply_blur(&self) -> WRequestResult<()>
130	{
131		/**
132		 * the `hyprland_surface_manager_v1` protocol already covers this, skip
133		 * <https://wayland.app/protocols/hyprland-surface-v1>
134		 */
135		match get_de() {
136			DE::Hyprland =>
137				return WRequestResult::Success(()),
138			_ => {}
139		}
140
141		WRequestResult::Fail(NotImplementedInCompositor)
142	}
143}

146impl Dispatch<wl_registry::WlRegistry, ()> for State
147{
148	fn event(
149		state: &mut Self,
150		registry: &wl_registry::WlRegistry,
151		event: wl_registry::Event,
152		_: &(),
153		_: &Connection,
154		qh: &QueueHandle<Self>,
155	) {
156		if let wl_registry::Event::Global { name, interface, .. } = event {
157			match &interface[..] {
158				"wl_compositor" => {
159					let compositor =
160						registry.bind::<wl_compositor::WlCompositor, _, _>(name, 1, qh, ());
161					let mut surface = compositor.create_surface(qh, ());
162					state.base_surface = Some(&mut surface as *mut wl_surface::WlSurface as *mut c_void);
163
164					//if state.wm_base.is_some() && state.xdg_surface.is_none() {
165					state.init_xdg_surface(qh);
166				}
167				"wl_seat" => {
168					registry.bind::<wl_seat::WlSeat, _, _>(name, 1, qh, ());
169				}
170				"xdg_wm_base" => {
171					let mut wm_base = registry.bind::<xdg_wm_base::XdgWmBase, _, _>(name, 1, qh, ());
172					state.wm_base =
173						Some(&mut wm_base as *mut xdg_wm_base::XdgWmBase as *mut c_void);
174
175					//let base_surface = state.base_surface as *mut wl_surface::WlSurface;
176					//let xdg_surface = state.xdg_surface as *mut wl_surface::WlSurface;
177
178					// I cant test if is Some or None here
179					// for now I will just pretend that xdg_surface doesn't exist
180					//if (*base_surface).is_alive() && (*xdg_surface).is_not_alive()
181					state.init_xdg_surface(qh);
182				}
183				_ => {}
184			}
185		}
186	}
187}

195impl Dispatch<xdg_wm_base::XdgWmBase, ()> for State {
196	fn event(
197		_: &mut Self,
198		wm_base: &xdg_wm_base::XdgWmBase,
199		event: xdg_wm_base::Event,
200		_: &(),

210impl Dispatch<xdg_surface::XdgSurface, ()> for State {
211	fn event(
212		state: &mut Self,
213		xdg_surface: &xdg_surface::XdgSurface,
214		event: xdg_surface::Event,
215		_: &(),
216		_: &Connection,
217		_: &QueueHandle<Self>,
218	) {
219		if let xdg_surface::Event::Configure { serial, .. } = event
220		{
221			xdg_surface.ack_configure(serial);
222			state.configured = true;
223			let surface = state.base_surface.unwrap() as *mut wl_surface::WlSurface;
224
225			if state.buffer.is_some() {
226				let buffer = state.buffer.unwrap() as *mut wl_buffer::WlBuffer;
227
228				unsafe { (*surface).attach(Some(&*buffer), 0, 0) };
229				unsafe { (*surface).commit() };
230			}
231		}
232	}
233}
